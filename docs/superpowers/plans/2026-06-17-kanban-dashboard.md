**Goal:** Build a read-only kanban dashboard that renders the project's slice status by parsing existing markdown (`docs/roadmap.md`, `openspec/changes/*/tasks.md`, `openspec/changes/archive/`), launched via a single cross-platform `uv run python tools/kanban.py` command.

**Architecture:** One Python file `tools/kanban.py` with three isolated units (parser, renderer, launcher) using only the stdlib. The parser reads markdown into a state dict; the renderer embeds that state as JSON inside a self-contained `index.html` (no `fetch`, works via `file://`); the launcher orchestrates and opens the browser via `webbrowser`. Tests live in `tests/unit/tools/` following the existing `tests/unit/` pattern.

**Tech Stack:** Python 3.12 stdlib only (`pathlib`, `re`, `html`, `webbrowser`, `json`, `datetime`), HTML/CSS/vanilla JS. No new dependencies. pytest for tests (already a dev dependency).

**Spec:** `docs/superpowers/specs/2026-06-17-kanban-dashboard-design.md`

def parse_state(repo_root: Path) -> dict:
    roadmap = (repo_root / "docs" / "roadmap.md").read_text(encoding="utf-8")
    milestones = _parse_milestones(roadmap)
    slices = _parse_slices(roadmap, milestones)
    return {
        "generated_at": datetime.now().astimezone().isoformat(timespec="seconds"),
        "milestones": milestones,
        "slices": slices,
    }


_MILESTONE_RE = re.compile(r"^##\s+(M\d+)\s+(.+?)\s*$", re.MULTILINE)


def _parse_milestones(roadmap_text: str) -> list[dict]:
    milestones = []
    for match in _MILESTONE_RE.finditer(roadmap_text):
        milestones.append(
            {"id": match.group(1), "title": f"{match.group(1)} {match.group(2)}", "closed_at": None}
        )
    return milestones


_SLICE_RE = re.compile(r"^\d+\.\s+`([^`]+)`:\s*(.+?)\s*$", re.MULTILINE)


def _parse_slices(roadmap_text: str, milestones: list[dict]) -> list[dict]:
    title_by_id = {m["id"]: m["title"] for m in milestones}
    current_milestone_id: str | None = None
    slices: list[dict] = []
    order_per_milestone: dict[str, int] = {}

    lines = roadmap_text.splitlines()
    for line in lines:
        m_header = _MILESTONE_RE.match(line)
        if m_header:
            current_milestone_id = m_header.group(1)
            continue
        m_slice = _SLICE_RE.match(line)
        if m_slice and current_milestone_id is not None:
            order_per_milestone[current_milestone_id] = (
                order_per_milestone.get(current_milestone_id, 0) + 1
            )
            slices.append(
                {
                    "id": m_slice.group(1),
                    "milestone_id": current_milestone_id,
                    "milestone_title": title_by_id.get(current_milestone_id, current_milestone_id),
                    "title": m_slice.group(2).rstrip("."),
                    "column": "backlog",
                    "total": 0,
                    "done": 0,
                    "has_spec": False,
                    "order": order_per_milestone[current_milestone_id],
                }
            )
    return slices
```

_ARCHIVE_MILESTONE_RE = re.compile(r"^(m\d+)")


def parse_state(repo_root: Path) -> dict:
    roadmap = (repo_root / "docs" / "roadmap.md").read_text(encoding="utf-8")
    milestones = _parse_milestones(roadmap)
    slices = _parse_slices(roadmap, milestones)
    _enrich_from_changes(repo_root, slices)
    _add_archived(repo_root, slices, milestones)
    _classify_all(slices)
    return {
        "generated_at": datetime.now().astimezone().isoformat(timespec="seconds"),
        "milestones": milestones,
        "slices": slices,
    }


def _enrich_from_changes(repo_root: Path, slices: list[dict]) -> None:
    for slice_ in slices:
        tasks_path = repo_root / "openspec" / "changes" / slice_["id"] / "tasks.md"
        slice_["has_spec"] = tasks_path.exists()
        total, done = _count_checkboxes(tasks_path)
        slice_["total"] = total
        slice_["done"] = done


def _add_archived(repo_root: Path, slices: list[dict], milestones: list[dict]) -> None:
    archive_dir = repo_root / "openspec" / "changes" / "archive"
    if not archive_dir.is_dir():
        return
    existing_ids = {s["id"] for s in slices}
    title_by_id = {m["id"]: m["title"] for m in milestones}
    for entry in sorted(archive_dir.iterdir()):
        if not entry.is_dir():
            continue
        # Directory names look like "2026-06-17-m1-foundation".
        suffix = entry.name
        m = _ARCHIVE_MILESTONE_RE.search(suffix)
        if m is None:
            continue
        milestone_id = m.group(1)
        # Slice id is everything after the leading date prefix, if present.
        slice_id = _strip_date_prefix(suffix)
        if slice_id in existing_ids:
            continue
        tasks_path = entry / "tasks.md"
        total, done = _count_checkboxes(tasks_path)
        slices.append(
            {
                "id": slice_id,
                "milestone_id": milestone_id,
                "milestone_title": title_by_id.get(milestone_id, milestone_id),
                "title": "(archivado)",
                "column": "done",
                "total": total,
                "done": done,
                "has_spec": tasks_path.exists(),
                "order": 999,
            }
        )


def _strip_date_prefix(name: str) -> str:
    # "2026-06-17-m1-foundation" -> "m1-foundation"; "m1-foundation" -> unchanged.
    m = re.match(r"^\d{4}-\d{2}-\d{2}-(.+)$", name)
    return m.group(1) if m else name


def _classify_all(slices: list[dict]) -> None:
    for slice_ in slices:
        if slice_["column"] == "done":
            continue  # archive entries are already done
        slice_["column"] = _classify(slice_)


def _classify(slice_: dict) -> str:
    total = slice_["total"]
    done = slice_["done"]
    if total > 0 and done == total:
        return "done"
    if slice_["has_spec"] and done > 0:
        return "in_progress"
    if slice_["has_spec"]:
        return "planned"
    return "backlog"
```
